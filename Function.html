def plot (data,subplot_row=2,subplot_col=2,plot_size=(18,10),param=None,y_lim=None,leg_label_1=None,leg_label_2=None,
          y_label=None,hspan_min=None,hspan_max=None,b_hline=-1,cross_vline=0,b_text=None,b_text_x=0,b_text_y=0,bd_text=None,
          bd_text_x=0,bd_text_y=0,who_text=None,who_hline=0,who_text_x=0,who_text_y=0,round_x=0,round_y=0,r1_text=None,
          r3_text=None,r5_text=None,r8_text=None,leg_col=2,leg_bbox=(.65, 1.05),leg_font=20):
    
    fig, x = plt.subplots(subplot_row,subplot_col,figsize=plot_size)

    a = x[0,0]
    a.plot(data.loc["C-01"][param].dropna(),"r--o",label=leg_label_1,linewidth=1.5,ms=15)
    a.plot(data.loc["C-02"][param].dropna(),"b--o",label=leg_label_2,linewidth=1.5,ms=15)
    a.set_ylim(y_lim)
    a.tick_params(labelsize=20)
    a.set_ylabel(y_label,fontsize=20)
    a.axhspan(ymin=hspan_min,ymax=hspan_max,facecolor='.7')
    a.axhline(b_hline,ls="--",c="k",linewidth=1.5)
    a.axhline(who_hline,ls="--",c="g",linewidth=1.5)
    a.axvline(cross_vline,ls="--",c="r",linewidth=1.5)
    a.text(bd_text_x,bd_text_y,bd_text, fontsize=16)
    a.text(who_text_x,who_text_y,who_text, fontsize=16)
    a.text(b_text_x,b_text_y,b_text, fontsize=16)
    a.text(round_x,round_y,r1_text, fontsize=20)

    b = x[0,1]
    b.plot(data.loc["C-05"][param].dropna(),"r--o",linewidth=1.5,ms=15)
    b.plot(data.loc["C-06"][param].dropna(),"b--o",linewidth=1.5,ms=15)
    b.set_ylim(y_lim)
    b.tick_params(labelsize=20)
    b.axhspan(ymin=hspan_min,ymax=hspan_max,facecolor='.7')
    b.axhline(b_hline,ls="--",c="k",linewidth=1.5)
    b.axhline(who_hline,ls="--",c="g",linewidth=1.5)
    b.text(bd_text_x,bd_text_y,bd_text, fontsize=16)
    b.text(who_text_x,who_text_y,who_text, fontsize=16)
    b.text(b_text_x,b_text_y,b_text, fontsize=16)
    b.text(round_x,round_y,r3_text, fontsize=20)

    c = x[1,0]
    c.plot(data.loc["C-09"][param].dropna(),"r--o",linewidth=1.5,ms=15)
    c.plot(data.loc["C-10"][param].dropna(),"b--o",linewidth=1.5,ms=15)
    c.set_ylim(y_lim)
    c.tick_params(labelsize=20)
    c.set_xlabel("V/Vi",fontsize=20)
    c.set_ylabel(y_label,fontsize=20)
    c.axhspan(ymin=hspan_min,ymax=hspan_max,facecolor='.7')
    c.axhline(b_hline,ls="--",c="k",linewidth=1.5)
    c.axhline(who_hline,ls="--",c="g",linewidth=1.5)
    c.text(bd_text_x,bd_text_y,bd_text, fontsize=16)
    c.text(who_text_x,who_text_y,who_text, fontsize=16)
    c.text(b_text_x,b_text_y,b_text, fontsize=16)
    c.text(round_x,round_y,r5_text, fontsize=20)

    d = x[1,1]
    d.plot(data.loc["C-15"][param].dropna(),"r--o",linewidth=1.5,ms=15)

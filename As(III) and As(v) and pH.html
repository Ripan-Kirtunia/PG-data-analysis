fig, x= plt.subplots(2,2,figsize=(18,13))
a = x[0,0]
a1 = a.twinx() # secondary y axis-1
a2 = a.twinx() # secondary y axis-2

a.plot(data.loc["C-01"]["Arsenic-3"].dropna(),"r--o",label="As(III)_(O2)",linewidth=1.5,ms=15)
a.plot(data.loc["C-02"]["Arsenic-3"].dropna(),"b--o",label="As(III)_(O2+NaOH)",linewidth=1.5,ms=15)
a.set_xlim([0,4])
a.set_ylim([0,90])
a.tick_params(labelsize=18)
a.set_ylabel("As (µg/L)",fontsize=24)
a.text(3,80,"Round-1", fontsize=24)

a1.plot(data.loc["C-01"]["Arsenic-5"].dropna(),"r--s",label="As(v)_regular",linewidth=1.5,ms=15)

a1.set_xlim([0,4])
a1.set_ylim([0,25])

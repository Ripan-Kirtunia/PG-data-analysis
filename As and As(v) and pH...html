fig, x= plt.subplots(4,1,figsize=(10,25))
a = x[0]
a1 = a.twinx() # secondary y axis-1
a2 = a.twinx() # secondary y axis-2

a.plot(data.loc["C-01"]["Total Arsenic"].dropna(),"r--o",label="As_reg",linewidth=1.5,ms=15)
a.plot(data.loc["C-02"]["Total Arsenic"].dropna(),"b--o",label="As_alt",linewidth=1.5,ms=15)
a.set_xlim([0,4])
a.set_ylim([10,100])
a.tick_params(labelsize=18)
a.set_xlabel("V/Vi",fontsize=20)
a.set_ylabel("As (g/L)",fontsize=20)
a.text(3,85,"Round-1", fontsize=20)

a1.plot(data.loc["C-01"]["Arsenic-5"].dropna(),"r--s",label="As(v)_reg",linewidth=1.5,ms=15)
a1.plot(data.loc["C-02"]["Arsenic-5"].dropna(),"b--s",label="As(v)_alt",linewidth=1.5,ms=15)
a1.set_xlim([0,4])
a1.set_ylim([1,25])
a1.tick_params(labelsize=18)

a2.plot(data.loc["C-01"]["pH"].dropna(),"r--d",label="pH_reg",linewidth=1.5,ms=15)
